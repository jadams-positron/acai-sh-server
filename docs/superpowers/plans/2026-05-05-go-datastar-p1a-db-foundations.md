# Go + Datastar Rewrite — Phase 1a: DB Foundations + serve subcommand

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Land the SQLite-backed `internal/store` package with goose migrations (full P1 schema from spec §4 squashed into one initial migration), sqlc wired up, and `acai serve` + `acai migrate` subcommands. End-state: `acai serve` boots an HTTP server with a working `/_health` endpoint that pings the DB; `acai migrate` runs all migrations against a target DB.

**Architecture:** Chi-routed HTTP server in `internal/server`. SQLite via `modernc.org/sqlite` opened with WAL + write/read pool split. Migrations embedded via `embed.FS` and run by `goose` library. sqlc generates typed query code from `internal/store/queries/*.sql` into `internal/store/sqlc/`.

**Tech Stack:** Go 1.26, `github.com/go-chi/chi/v5`, `modernc.org/sqlite`, `github.com/pressly/goose/v3`, `github.com/sqlc-dev/sqlc` (codegen tool, not runtime), `log/slog`.

**Reference:** Spec at `docs/superpowers/specs/2026-05-05-go-datastar-rewrite-design.md` (sections 2, 3, 4, 8). P0 plan at `docs/superpowers/plans/2026-05-05-go-datastar-p0-scaffolding.md` (already executed; branch `rewrite/go-datastar` is at `86ff84c`).

```go
func Load() (*Config, error) {
	cfg := &Config{
		LogLevel:     getenvDefault("LOG_LEVEL", "info"),
		DatabasePath: getenvDefault("DATABASE_PATH", "./acai.db"),
	}
	switch cfg.LogLevel {
	case "debug", "info", "warn", "error":
		// ok
	default:
		return nil, fmt.Errorf("config: invalid LOG_LEVEL %q (allowed: debug, info, warn, error)", cfg.LogLevel)
	}

	httpPort, err := strconv.Atoi(getenvDefault("HTTP_PORT", "4000"))
	if err != nil {
		return nil, fmt.Errorf("config: invalid HTTP_PORT: %w", err)
	}
	if httpPort < 1 || httpPort > 65535 {
		return nil, fmt.Errorf("config: HTTP_PORT %d out of range [1, 65535]", httpPort)
	}
	cfg.HTTPPort = httpPort

	return cfg, nil
}
```

```go
// Package store owns SQLite connectivity and provides the typed query layer
// (sqlc-generated) plus a small helper API for transactions.
//
// SQLite single-writer semantics: we open two *sql.DB handles against the same
// file. The Write pool is capped to one connection (serializes writers,
// eliminates SQLITE_BUSY retries); the Read pool allows many concurrent
// readers (WAL makes this safe).
package store

import (
	"database/sql"
	"errors"
	"fmt"
	"net/url"
	"runtime"

	_ "modernc.org/sqlite" // pure-Go SQLite driver, registered as "sqlite"
)

// DB is the dual-pool SQLite handle. Use Read for queries, Write for mutations
// (or for a transaction that needs to mutate).
type DB struct {
	Read  *sql.DB
	Write *sql.DB
	path  string
}

// Open returns a *DB pointing at the SQLite file at path. The file is created
// if it doesn't exist. PRAGMAs are applied to both pools; they are
// connection-scoped, but we re-apply on every new connection by issuing them
// in the DSN.
func Open(path string) (*DB, error) {
	if path == "" {
		return nil, errors.New("store: Open requires a non-empty path")
	}

	dsn := buildDSN(path)

	read, err := sql.Open("sqlite", dsn)
	if err != nil {
		return nil, fmt.Errorf("store: open read pool: %w", err)
	}
	read.SetMaxOpenConns(runtime.NumCPU() * 2)

	write, err := sql.Open("sqlite", dsn)
	if err != nil {
		_ = read.Close()
		return nil, fmt.Errorf("store: open write pool: %w", err)
	}
	write.SetMaxOpenConns(1)

	// Verify both pools can connect. Defer a close on failure so we don't leak.
	if err := read.Ping(); err != nil {
		_ = read.Close()
		_ = write.Close()
		return nil, fmt.Errorf("store: ping read pool: %w", err)
	}
	if err := write.Ping(); err != nil {
		_ = read.Close()
		_ = write.Close()
		return nil, fmt.Errorf("store: ping write pool: %w", err)
	}

	return &DB{Read: read, Write: write, path: path}, nil
}

// Close releases both pools. Safe to call multiple times.
func (db *DB) Close() error {
	if db == nil {
		return nil
	}
	var firstErr error
	if db.Read != nil {
		if err := db.Read.Close(); err != nil {
			firstErr = err
		}
	}
	if db.Write != nil {
		if err := db.Write.Close(); err != nil && firstErr == nil {
			firstErr = err
		}
	}
	return firstErr
}

// Path returns the filesystem path the DB was opened from.
func (db *DB) Path() string { return db.path }

// buildDSN constructs the modernc.org/sqlite DSN with PRAGMAs encoded as
// _pragma= query parameters. modernc applies these on every new connection.
func buildDSN(path string) string {
	q := url.Values{}
	q.Add("_pragma", "journal_mode(WAL)")
	q.Add("_pragma", "synchronous(NORMAL)")
	q.Add("_pragma", "foreign_keys(ON)")
	q.Add("_pragma", "busy_timeout(5000)")
	q.Add("_pragma", "temp_store(MEMORY)")
	q.Add("_pragma", "cache_size(-20000)")
	return "file:" + path + "?" + q.Encode()
}
```

```go
package store

import (
	"context"
	"fmt"

	"github.com/pressly/goose/v3"

	"github.com/acai-sh/server/internal/store/migrations"
)

// RunMigrations applies all goose Up migrations embedded in
// internal/store/migrations against the write pool. Idempotent.
//
// The migrations run on the write pool because they are mutating, and goose
// holds a single connection for the duration of an Up.
func RunMigrations(ctx context.Context, db *DB) error {
	if db == nil || db.Write == nil {
		return fmt.Errorf("store: RunMigrations requires an open *DB")
```

```go
package ops

import (
	"encoding/json"
	"net/http"

	"github.com/acai-sh/server/internal/store"
)

// HealthHandler returns an http.Handler that responds to GET /_health with a
// JSON body indicating service and DB status. Writes 200 if the DB is
// reachable, 503 otherwise.
//
// Response shape (success):
//
//	{"status":"ok","db":"ok","version":"<v>"}
//
// Response shape (DB unreachable):
//
//	{"status":"degraded","db":"error","version":"<v>","error":"<msg>"}
func HealthHandler(db *store.DB, version string) http.Handler {
	return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
		w.Header().Set("Content-Type", "application/json")

		body := map[string]string{
			"status":  "ok",
			"db":      "ok",
			"version": version,
		}
		code := http.StatusOK

		if err := db.Read.PingContext(r.Context()); err != nil {
			body["status"] = "degraded"
			body["db"] = "error"
			body["error"] = err.Error()
			code = http.StatusServiceUnavailable
		}

		w.WriteHeader(code)
		_ = json.NewEncoder(w).Encode(body)
	})
}
```

```go
// Package server owns the HTTP listener and chi router lifecycle. Subsequent
// phases extend this with API and site route groups; P1a wires only /_health.
package server

import (
	"context"
	"errors"
	"fmt"
	"log/slog"
	"net"
	"net/http"
	"strconv"
	"time"

	"github.com/acai-sh/server/internal/config"
	"github.com/acai-sh/server/internal/store"
)

// Server is the HTTP server bundle.
type Server struct {
	cfg     *config.Config
	logger  *slog.Logger
	db      *store.DB
	version string
	http    *http.Server
}

// New constructs a *Server with all dependencies wired. The DB is expected to
// already be migrated.
func New(cfg *config.Config, logger *slog.Logger, db *store.DB, version string) (*Server, error) {
	if cfg == nil {
		return nil, errors.New("server: cfg is nil")
	}
	if logger == nil {
		return nil, errors.New("server: logger is nil")
	}
	if db == nil {
		return nil, errors.New("server: db is nil")
	}

	router := newRouter(db, version)

	httpServer := &http.Server{
		Handler:           router,
		ReadHeaderTimeout: 10 * time.Second,
		ReadTimeout:       30 * time.Second,
		WriteTimeout:      30 * time.Second,
		IdleTimeout:       60 * time.Second,
	}

	return &Server{
		cfg:     cfg,
		logger:  logger,
		db:      db,
		version: version,
		http:    httpServer,
	}, nil
}

// Run starts the HTTP listener on cfg.HTTPPort (or an ephemeral port if 0)
// and blocks until ctx is canceled. The actual bound address is sent on addrCh
// once the listener is up; tests use this for ephemeral-port discovery.
//
// On ctx cancellation, Run gracefully shuts down with a 10s timeout.
func (s *Server) Run(ctx context.Context, addrCh chan<- string) error {
	addr := net.JoinHostPort("0.0.0.0", strconv.Itoa(s.cfg.HTTPPort))
	ln, err := net.Listen("tcp", addr)
	if err != nil {
		return fmt.Errorf("server: listen %s: %w", addr, err)
	}

	if addrCh != nil {
		select {
		case addrCh <- ln.Addr().String():
		default:
		}
	}

	s.logger.Info("server starting", slog.String("addr", ln.Addr().String()), slog.String("version", s.version))

	errCh := make(chan error, 1)
	go func() {
		if err := s.http.Serve(ln); err != nil && !errors.Is(err, http.ErrServerClosed) {
			errCh <- err
			return
		}
		errCh <- nil
	}()

	select {
	case <-ctx.Done():
		s.logger.Info("server stopping")
		shutdownCtx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
		defer cancel()
		if err := s.http.Shutdown(shutdownCtx); err != nil {
			return fmt.Errorf("server: shutdown: %w", err)
		}
		<-errCh
		return nil
	case err := <-errCh:
		return err
	}
}
```

**Placeholder scan:** No "TBD" or "implement later" — every step has the actual code or command to run.

**Risk register (this phase only):**
- **`commit` reserved word in SQLite** — handled by renaming to `commit_hash` in `feature_branch_refs`. Document in spec amendments later.
- **goose migration `-- +goose StatementEnd` placement** — the migration uses StatementBegin/End around blocks of multi-statement SQL because some of the CREATE TABLE statements are inside the same block. Verify after T2 that goose accepts them.
- **modernc.org/sqlite PRAGMA-via-DSN** — `_pragma=` query param is the documented way; T3's tests verify they actually applied (e.g., `journal_mode = wal`).
- **Test timing flakes on T7's server test** — the server uses `signal.NotifyContext` in main but tests use `context.WithCancel`; the listener is up before `addrCh <-` so race is bounded.
